# 1. Start Kafka (fresh)
docker run -d \
  --name kafka-local \
  -p 9092:9092 \
  -e KAFKA_NODE_ID=1 \
  -e KAFKA_PROCESS_ROLES=broker,controller \
  -e KAFKA_LISTENERS=PLAINTEXT://0.0.0.0:9092,CONTROLLER://0.0.0.0:9093 \
  -e KAFKA_ADVERTISED_LISTENERS=PLAINTEXT://localhost:9092 \
  -e KAFKA_CONTROLLER_QUORUM_VOTERS=1@localhost:9093 \
  -e KAFKA_LISTENER_SECURITY_PROTOCOL_MAP=PLAINTEXT:PLAINTEXT,CONTROLLER:PLAINTEXT \
  -e KAFKA_CONTROLLER_LISTENER_NAMES=CONTROLLER \
  -e KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR=1 \
  -e KAFKA_LOG_DIRS=/var/lib/kafka/data \
  apache/kafka:3.7.0

# 2. Activate venv + load PATs (in the streaming/ dir)
cd streaming
source .venv/bin/activate
set -a && source .env && set +a

# 3. Create topics (once)
python stream02_create_topics.py

# 4. Run producer (Terminal 1)
python stream03_producer.py

# 5. Run schema test (Terminal 2, after producer publishes some events)
python stream04_schema_test.py

# To stop everything when done
docker stop kafka-local && docker rm kafka-local
